def send_message(chat_id, text):
    url = f"{BASE_URL}/sendMessage"
    data = {
        'chat_id': chat_id,
        'text': text
    }

    data = json.dumps(data).encode('utf-8')
    req = urllib.request.Request(
        url,
        data=data,
        headers={'Content-Type': 'application/json'}
    )

        with urllib.request.urlopen(req) as response:
